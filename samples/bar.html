with a fill color function

config: 
(empty)
dataset fillColor: 
function(dataValue,animationPercent){
	// dataValue is the raw data value. Not guaranteed to be in any
	// range, but luckily we have access to the raw data input just
	// below this.
	// animationPercent is how far along in the animation we are,
	// (in case you want to vary the bar color with animation)
	
	// hue is [0,360]
	var hue = Math.round(dataValue*animationPercent/40*360);
	// saturation is [0,100]
	var saturation = 100;
	// lightness is [0,100]
	var lightness = 50;
	var color = "hsla("+hue+", "+saturation+"%, "+lightness+"%, 1)";
	// Make sure to handle the case where one or both of them is NaN:
	// this is a side-effect of the bar-with-errors update
	if (isNaN(dataValue) || isNaN(animationPercent)) {
		color = "hsla(0, 100%, 50%, 1)";
	}
	return color;
}
